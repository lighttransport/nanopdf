# Pluggable canvas backends. Matching CMake's LVG_ENABLE_BLEND2D / _THORVG;
# both are OFF by default so the common build path has no external deps.
#
# Blend2D has no native Meson build; enabling this option expects a
# prebuilt static library (see scripts/fetch-verify-deps.sh). If a
# turnkey Meson workflow matters to you, use the CMake build instead.
option('lui_enable_blend2d',
  type        : 'feature',
  value       : 'disabled',
  description : 'Enable the blend2d canvas backend (requires prebuilt ref/blend2d)',
)

option('lui_enable_thorvg',
  type        : 'feature',
  value       : 'disabled',
  description : 'Enable the thorvg canvas backend (requires prebuilt ref/thorvg/build)',
)
